# wsl2-backup rclone filter rules
# Rules are applied top-to-bottom. First match wins.
# Syntax: - exclude, + include

# ── Go ────────────────────────────────────────────────────────────────────────
- /go/pkg/**
- /go/bin/**
- vendor/**
- *.o
- *.a

# ── Node / TypeScript ─────────────────────────────────────────────────────────
- node_modules/**
- .npm/**
- .yarn/cache/**
- .yarn/unplugged/**
- .pnpm-store/**
- dist/**
- .next/**
- .nuxt/**
- .svelte-kit/**
- out/**
- coverage/**

# ── .NET / C# ─────────────────────────────────────────────────────────────────
- **/obj/**
- **/bin/Debug/**
- **/bin/Release/**
- *.user
- .vs/**
- .vscode/.browse.VC.db*
- TestResults/**

# ── Python (toolchain overlap) ────────────────────────────────────────────────
- __pycache__/**
- *.pyc
- *.pyo
- .venv/**
- venv/**
- .tox/**

# ── Rust (if ever used) ───────────────────────────────────────────────────────
- target/debug/**
- target/release/**

# ── Docker ────────────────────────────────────────────────────────────────────
- .docker/**
# Docker volume data — owned by container users (root/UID 999), unreadable by
# the host user. These are runtime state, not source files worth backing up.
- **/data/mongodb-*/**
- **/structurizr/1/**
- **/structurizr/index/**
- **/structurizr/logs/**

# ── General build & cache ────────────────────────────────────────────────────
- build/**
- .cache/**
- tmp/**
- temp/**
- *.log
- logs/**

# ── Large VCS objects ─────────────────────────────────────────────────────────
- **/.git/lfs/**

# ── OS & editor junk ──────────────────────────────────────────────────────────
- .DS_Store
- Thumbs.db
- *.swp
- *.swo
- *~

# ── Large binaries / archives (installers, disk images — NOT database dumps) ──
- *.iso
- *.img
- *.vmdk
- *.deb
- *.rpm
- *.AppImage

# ── Compressed archives (exclude generic ones, keep db dumps) ─────────────────
# If you store database dumps as .tar.gz or .zip, comment these out
- *.zip
- *.tar.gz

# ── Include everything else ───────────────────────────────────────────────────
+ **
